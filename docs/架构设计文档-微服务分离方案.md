## 五、服务间通信

### 5.1 Java → Python (HTTP调用)

```java
// Java后端调用Python AI服务示例
@RestController
@RequestMapping("/api/v1/ai")
public class AIController {

    @Value("${ai.service.url:http://python-ai:8000}")
    private String aiServiceUrl;

    @PostMapping("/vector-search")
    public Result<List<Image>> vectorSearch(@RequestParam Long imageId) {
        // 1. 获取图片信息
        Image image = imageService.getById(imageId);

        // 2. 调用Python AI服务进行向量搜索
        String pythonUrl = aiServiceUrl + "/api/v1/ai/vector/search";
        List<Long> similarIds = restTemplate.postForObject(pythonUrl,
            Map.of("imageUrl", image.getUrl()),
            List.class);

        // 3. 返回相似图片
        return Result.success(imageService.listByIds(similarIds));
    }
}
```

### 5.2 Python → Java (获取业务数据)

```python
# Python AI服务调用Java后端获取数据
import httpx

class JavaAPIClient:
    def __init__(self, base_url: str = "http://java-backend:8080"):
        self.base_url = base_url
        self.token = None

    def authenticate(self, username: str, password: str):
        """获取访问令牌"""
        response = httpx.post(f"{self.base_url}/api/v1/auth/login", json={
            "username": username,
            "password": password
        })
        self.token = response.json()["data"]["accessToken"]

    def get_products(self, category: str = None):
        """获取产品列表用于向量化"""
        headers = {"Authorization": f"Bearer {self.token}"}
        response = httpx.get(
            f"{self.base_url}/api/v1/products",
            params={"category": category} if category else None,
            headers=headers
        )
        return response.json()["data"]["items"]

    def get_images(self, product_id: int):
        """获取产品图片"""
        headers = {"Authorization": f"Bearer {self.token}"}
        response = httpx.get(
            f"{self.base_url}/api/v1/images",
            params={"productId": product_id},
            headers=headers
        )
        return response.json()["data"]["items"]
```
