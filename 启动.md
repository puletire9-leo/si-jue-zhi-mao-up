 完整构建命令（按顺序）：
  
  # 1. 构建公共 Python 基础镜像
  docker compose -f docker-compose.base.yml build

  # 2. 启动公共基础设施（MySQL + Redis）
  docker compose -f docker-compose.yml up -d

  # 3. 等待 MySQL 就绪（30 秒左右），导入数据
  cmd /c "(echo USE sijuelishi; && type backend\database\backup\sijuelishi_full_20260515_160545.sql) | docker exec -i    
  sjzm-mysql mysql -uroot -proot -P3307"

  # 4. 创建开发数据库
  docker exec sjzm-mysql mysql -uroot -proot -P3307 -e "CREATE DATABASE IF NOT EXISTS sijuelishi_dev CHARACTER SET       
  utf8mb4 COLLATE utf8mb4_unicode_ci; GRANT ALL PRIVILEGES ON sijuelishi_dev.* TO 'sijue'@'%'; FLUSH PRIVILEGES;"        

  # 5. 启动开发环境
  docker compose -f docker-compose.yml -f docker-compose.dev.yml up -d --build

  # 6. 启动生产环境（和开发同时跑）
  docker compose -f docker-compose.yml -f docker-compose.prod.yml up -d --build
